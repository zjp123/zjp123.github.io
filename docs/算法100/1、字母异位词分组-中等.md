var groupAnagrams = function(strs) {
    if (strs.length <=1 ) return [strs]
    const arr = []
    const mapstr = {}
    
    for(let i = 0; i<strs.length; i++) {
        const temparr = strs[i].split('').sort().join()
        if (!mapstr[temparr]) {
            mapstr[temparr] = []
        } 
            mapstr[temparr].push(strs[i])

        
        
    }
    Object.values(mapstr).forEach((val) => {
        arr.push(val)
    })
    return arr
};
